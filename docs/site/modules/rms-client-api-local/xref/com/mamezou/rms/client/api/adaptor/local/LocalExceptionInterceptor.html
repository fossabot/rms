3   import javax.annotation.Priority;
4   import javax.interceptor.AroundInvoke;
5   import javax.interceptor.Interceptor;
6   import javax.interceptor.InvocationContext;
7   
8   @Interceptor
9   @Priority(Interceptor.Priority.APPLICATION)
10  @HandleExceptions
11  public class LocalExceptionInterceptor {
12  
13      @AroundInvoke
14      public Object obj(InvocationContext ic) {
15          Object result = null;
16          try {
17              result = ic.proceed();
18          } catch (RuntimeException e) {
19              LocalExceptionHandler.throwConvertedException(e);
20          } catch (Exception e) {
21              LocalExceptionHandler.throwConvertedException(e);
22          }
23          return result;
24      }
25  }
