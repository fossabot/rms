16  import com.mamezou.rms.client.console.ui.textio.TextIoUtils;
17  
18  import lombok.RequiredArgsConstructor;
19  
20  @RequiredArgsConstructor
21  public class EditUserScreen implements RmsScreen {
22  
23      private final RentalReservationClientApi clientApi;
24  
25      @Override
26      public Transition play(UserAccountClientDto loginUser, boolean printHeader) {
27  
28          if (printHeader) {
29              printScreenHeader(loginUser, "ユーザ情報編集画面");
30          }
31  
32          // ユーザ一覧を表示
33          var users = clientApi.getAllUserAccounts();
34          println(EDIT_USER_INFORMATION);
35          users.forEach(dto ->
36              println(ClientConstants.USER_ACCOUNT_FORMAT.format(dto))
37          );
38          blankLine();
39  
40          // 編集するユーザを選択
41          var selectNumber = newIntInputReader()
42                  .withSelectableValues(
43                          users.stream()
44                              .map(UserAccountClientDto::getId)
45                              .collect(Collectors.toList()),
46                          SCREEN_BREAK_VALUE)
47                  .read("ユーザ番号");
48          if (TextIoUtils.isBreak(selectNumber)) {
49              return Transition.ADMIN_MAIN;
50          }
51          blankLine();
52  
53          var targetUser = users.stream()
54                  .filter(user -> user.getId().equals(selectNumber))
55                  .findFirst()
56                  .get();
57  
58          // パスワードの入力
59          var password = newStringInputReader()
60                  .withDefaultValue(targetUser.getPassword())
61                  .withMinLength(5)
62                  .withMaxLength(15)
63                  .read("パスワード");
64          targetUser.setPassword(password);
65  
66          // ユーザ名の入力
67          var userName = newStringInputReader()
68                  .withDefaultValue(targetUser.getUserName())
69                  .withMinLength(1)
70                  .read("ユーザ名");
71          targetUser.setUserName(userName);
72  
73          // 電話番号の入力
74          var phoneNumber = newStringInputReader()
75                  .withDefaultValue(targetUser.getPhoneNumber())
76                  .withMaxLength(14)
77                  .withPattern(PatternMessage.PHONE_NUMBER)
78                  .read("電話番号（省略可）");
79          targetUser.setPhoneNumber(phoneNumber);
80  
81          // 連絡先の入力
82          var contact = newStringInputReader()
83                  .withDefaultValue(targetUser.getPhoneNumber())
84                  .withMaxLength(15)
85                  .read("連絡先（省略可）");
86          targetUser.setContact(contact);
87  
88          // 連絡先の入力
89          var userType = newEnumInputReader(ClientUserType.class)
90                  .withDefaultValue(targetUser.getUserType())
91                  .read("権限");
92          targetUser.setUserType(userType);
93  
94          // ユーザ情報の更新実行
95          try {
96              var updatedUser = clientApi.updateUserAccount(targetUser);
97              printResultInformation(updatedUser);
98              return Transition.ADMIN_MAIN;
99  
100         } catch (BusinessFlowClientException e) {
101             printServerError(e);
102             return play(loginUser, false); // start over!!
103 
104         }
105     }
106 
107     private void printResultInformation(UserAccountClientDto updatedUserAccount) {
108         blankLine();
109         println("***** ユーザ登録結果 *****");
110         printf("[%s]のユーザ情報を更新しました", updatedUserAccount.getId());
111         blankLine();
112         blankLine();
113         waitPressEnter();
114     }
115 }
