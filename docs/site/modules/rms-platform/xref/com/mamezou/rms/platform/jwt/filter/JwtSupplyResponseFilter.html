3   import java.io.IOException;
4   
5   import javax.inject.Inject;
6   import javax.ws.rs.ConstrainedTo;
7   import javax.ws.rs.RuntimeType;
8   import javax.ws.rs.container.ContainerRequestContext;
9   import javax.ws.rs.container.ContainerResponseContext;
10  import javax.ws.rs.container.ContainerResponseFilter;
11  import javax.ws.rs.core.HttpHeaders;
12  
13  import org.apache.commons.lang3.Range;
14  
15  import com.mamezou.rms.platform.jwt.JsonWebTokenGenerator;
16  import com.mamezou.rms.platform.jwt.JsonWebTokenGenerator.UserClaims;
17  import com.mamezou.rms.platform.jwt.JsonWebTokenGenerator.UserClaimsFactory;
18  import com.mamezou.rms.platform.jwt.JwtConfig;
19  
20  import lombok.extern.slf4j.Slf4j;
21  
22  @GenerateToken
23  @ConstrainedTo(RuntimeType.SERVER)
24  @Slf4j
25  public class JwtSupplyResponseFilter implements ContainerResponseFilter {
26  
27      private static Range<Integer> SUCCESS_STATUS = Range.between(200, 299);
28  
29      private UserClaimsFactory userClaimsFactory;
30      private JsonWebTokenGenerator tokenGenerator;
31  
32      @Inject
33      public JwtSupplyResponseFilter(UserClaimsFactory factory, JsonWebTokenGenerator generator) {
34          this.userClaimsFactory = factory;
35          this.tokenGenerator = generator;
36      }
37  
38      @Override
39      public void filter(ContainerRequestContext requestContext, ContainerResponseContext responseContext) throws IOException {
40  
41          // 例外が発生した場合はExecptionMapperでレスポンスが設定されているので
42          // まずはHTTPステータスを見て成功か失敗かを判定
43          if (!SUCCESS_STATUS.contains(responseContext.getStatus())) {
44              return;
45          }
46  
47          if (!responseContext.hasEntity()) {
48              log.warn("ボディが設定されていません");
49              return;
50          }
51  
52          Object entity = responseContext.getEntity();
53          if (!userClaimsFactory.canNewInstanceFrom(entity)) {
54              log.warn("ボディのインスタンスが想定外です [class={}]", entity.getClass().getName());
55              return;
56          }
57  
58          // JwtTokenの生成
59          UserClaims userClaims = userClaimsFactory.newInstanceFrom(entity);
60          String jwtToken = tokenGenerator.generateToken(userClaims);
61          log.info("Generated JWT-Token=>[{}]", jwtToken); // ホントはログに書いちゃダメだけどネ
62  
63          var headers = responseContext.getHeaders();
64          headers.add(HttpHeaders.AUTHORIZATION, JwtConfig.BEARER_MARK + " " + jwtToken);
65      }
66  }
