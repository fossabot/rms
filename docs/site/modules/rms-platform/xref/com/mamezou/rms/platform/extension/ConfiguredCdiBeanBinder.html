62      private List<AliasKey> seekAliasKeys(String prefix) {
63          // (config)
64          // a.b.c.x.class = Foo
65          // a.b.c.x.scope = application
66          // a.b.c.y.tyepe = Bar
67          // a.b.c.y.scope = request
68          // ---------------------------
69          // keyPrefix => "a.b.c"
70          // map       => ["a.b.c.x", "a.b.c.x", "a.b.c.y", "a.b.c.y"]
71          // distinct  => ["a.b.c.x", "a.b.c.y"]
72          // map       => [{"x":"a.b.c.x"}, {"y":"a.b.c.y"}]
73          //
74          return StreamSupport.stream(config.getPropertyNames().spliterator(), false)
75                  .filter(s -> s.startsWith(prefix))
76                  .map(s -> s.substring(0, prefix.length() + 1 + s.substring(prefix.length() + 1).indexOf('.')))
77                  .distinct()
78                  .map(s -> new AliasKey(s.substring(prefix.length() + 1), s))
79                  .collect(Collectors.toList());
80      }

85      public class ConfiguredCdiBeanBindFinisher {
86  
87          private ConfiguredCdiBeanBindFinisher() {
88              // nop
89          }
90  
91          public ConfiguredCdiBean bind() {
92  
93              Optional<String> alias = config.getOptionalValue(registerPrefix + ".alias", String.class);
94              if (alias.isPresent()) {
95                  if (aliasKeys == null) {
96                      throw new IllegalStateException("alias() not called!");
97                  }
98                  AliasKey aliasKey = aliasKeys.stream()
99                          .filter(key -> alias.get().equals(key.aliasName))
100                         .findFirst()
101                         .orElseThrow(() -> new IllegalArgumentException("alias is unmatch. [" + alias.get() + "]"));
102                 return ConfiguredCdiBeanBinder.newBinder(config).key(aliasKey.aliasConfigKey).bind();
103             }
104 
105             var configCdiBean = new ConfiguredCdiBean();
106             configCdiBean.beanClass = config.getOptionalValue(registerPrefix + ".class", Class.class)
107                     .orElseThrow(() -> new IllegalArgumentException("class is required. [" + registerPrefix + ".class]")); // class is required.
108 
109             String scopeValue = config.getOptionalValue(registerPrefix + ".scope", String.class).orElse("application"); // applicaiton is default.
110             configCdiBean.scope = Optional.ofNullable(BEAN_SCOPES.get(scopeValue))
111                     .orElseThrow(() -> new IllegalArgumentException("unknown scoped value. [" + registerPrefix + ".scope:" + scopeValue + "]"));
112 
113             configCdiBean.id = config.getOptionalValue(registerPrefix + ".id", String.class).orElse(configCdiBean.beanClass.getName()); // FQCN is default.
114 
115             return configCdiBean;
116         }

143     public class ConfiguredCdiBeansListBindFinisher {
144 
145         private ConfiguredCdiBeansListBinder parent;
146 
147         private ConfiguredCdiBeansListBindFinisher(ConfiguredCdiBeansListBinder listBinder) {
148             this.parent = listBinder;
149         }
150 
151         public List<ConfiguredCdiBean> bind() {
152 
153             List<AliasKey> foundAliasKeys = List.of();
154             if (parent.listAliasPrefix != null) {
155                 foundAliasKeys = seekAliasKeys(parent.listAliasPrefix);
156             }
157 
158             // (by single config)
159             // a.b.c.class = Foo
160             // a.b.c.scope = application
161             //
162             Optional<String> clazz = config.getOptionalValue(parent.listRegisterPrefix + ".class", String.class);
163             Optional<String> alias = config.getOptionalValue(parent.listRegisterPrefix + ".alias", String.class);
164             if (clazz.isPresent() || alias.isPresent()) {
165                 ConfiguredCdiBean configCdiBean =
166                         ConfiguredCdiBeanBinder.newBinder(config)
167                             .aliasKeys(foundAliasKeys)
168                             .key(parent.listRegisterPrefix)
169                             .bind();
170                 return List.of(configCdiBean);
171             }
172 
173             // (by list config)
174             // a.b.c.0.class = Foo
175             // a.b.c.0.scope = application
176             // a.b.c.1.tyepe = Bar
177             // a.b.c.1.scope = request
178             // ---------------------------
179             // keyPrefix => "a.b.c"
180             // map       => ["a.b.c.0", "a.b.c.0", "a.b.c.1", "a.b.c.1"]
181             // distinct  => ["a.b.c.0", "a.b.c.1"]
182             //
183             List<String> keys = StreamSupport.stream(config.getPropertyNames().spliterator(), false)
184                     .filter(s -> s.startsWith(parent.listRegisterPrefix))
185                     .map(s -> s.substring(0, parent.listRegisterPrefix.length() + 1 + s.substring(parent.listRegisterPrefix.length() + 1).indexOf('.')))
186                     .distinct()
187                     .collect(Collectors.toList());
188 
189             List<ConfiguredCdiBean> configCdiBeans = new ArrayList<>();
190             for (int i = 0; i < keys.size(); i++) {
191                 configCdiBeans.add(ConfiguredCdiBeanBinder.newBinder(config).aliasKeys(foundAliasKeys).key(parent.listRegisterPrefix + "." + i).bind());
192             }
193             return configCdiBeans;
194         }
195     }
