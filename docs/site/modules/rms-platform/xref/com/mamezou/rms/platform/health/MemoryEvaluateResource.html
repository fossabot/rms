10  import org.eclipse.microprofile.openapi.annotations.Operation;

12  import org.eclipse.microprofile.openapi.annotations.media.Schema;
13  import org.eclipse.microprofile.openapi.annotations.parameters.Parameter;

15  import org.eclipse.microprofile.openapi.annotations.responses.APIResponse;
16  import org.eclipse.microprofile.openapi.annotations.tags.Tag;
17  
18  import com.mamezou.rms.platform.health.MemoryHealthCheck.MemoryLivenessEvaluator;
19  
20  import lombok.extern.slf4j.Slf4j;
21  
22  @Path("/memeval")
23  @ApplicationScoped
24  @Slf4j
25  public class MemoryEvaluateResource {
26  
27      @Inject
28      private Event<MemoryLivenessEvaluator> event;
29  
30      @GET
31      @Tag(name = "SeverManagement API")
32      @Operation(
33          operationId = "resetEvaluateMethod",
34          summary = "memory-health-check-livenessのチェック方法の変更",
35          description = "/health/readyによるReadinessProbeのmemory-health-check-livenessのチェック方法を変更する")
36      @Parameters({
37          @Parameter(name = "method", description = "評価方法。abs:絶対評価, rel:相対評価", required = true,
38              schema = @Schema(implementation = String.class)),
39          @Parameter(name = "val",  description = "閾値。絶対評価の場合は使用ヒープサイズ, 相対評価の場合はヒープの使用率、", required = true,
40              schema = @Schema(implementation = Long.class))
41      })
42      @APIResponse(
43          responseCode = "200",
44          description = "成功。常に\"accepted.\"を返す",
45          content = @Content(mediaType = "application/text"))
46      public String resetEvaluateMethod(@QueryParam("method") String method, @QueryParam("val") long val){
47          log.info("reset memory evaluate method. methdo={}, val={}", method, val);
48          event.fire(MemoryLivenessEvaluator.of(method, val));
49          return "accepted.";
50      }
51  }
