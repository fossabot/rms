12  /**
13   * 設定によりCDI化されることを表すアノテーション。
14   * 設定ファイルによりCDI登録するクラスは<code>@ApplicationScoped</code>などのCDIであることを
15   * 明示的に示すアノテーションが付けないため、一見してCDIBeanかが分かりずらい。<br>
16   * 設定によりCDI化されるクラスにはこのアノテーションを付けることでCDIBeanであることを分かり易く
17   * する。<br>
18   * なお、コードを分かりやすくすることのみが目的のアノテーションのため、実行時の動作にはなんら
19   * 影響せず、このアノテーションが付いていなくても設定ファイルに登録されているクラスは問題なく
20   * CDI化が行われる。
21   *
22   * @see ApplicationInitializerCdiExtension#registerConfiguredBeans(BeforeBeanDiscovery)
23   */
24  @Documented
25  @Retention(SOURCE)
26  @Target({ TYPE })
27  public @interface ConfiguableScoped {
28  }
