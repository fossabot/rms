1   package com.mamezou.rms.platform.role;
2   
3   import javax.annotation.security.DenyAll;
4   import javax.annotation.security.RolesAllowed;
5   import javax.ws.rs.container.DynamicFeature;
6   import javax.ws.rs.container.ResourceInfo;
7   import javax.ws.rs.core.FeatureContext;
8   
9   import org.eclipse.microprofile.config.ConfigProvider;
10  
11  import lombok.extern.slf4j.Slf4j;
12  
13  @Slf4j
14  public class RoleSecurityDynamicFeature implements DynamicFeature {
15  
16      @Override
17      public void configure(ResourceInfo resourceInfo, FeatureContext context) {
18  
19          if (!enableFilter()) {
20              log.debug("設定(jwt.filter.enable)がOFFになっているためロールチェックは行いません");
21              return;
22          }
23  
24          // methodのアノテーションから優先して確認
25          var method = resourceInfo.getResourceMethod();
26          if (method.getAnnotation(DenyAll.class) != null) {
27              context.register(new DenyAllRequestFilter());
28              return;
29          }
30          RolesAllowed  rolesAllowed = method.getAnnotation(RolesAllowed.class);
31          if (rolesAllowed != null) {
32              context.register(new RolesAllowedRequestFilter(rolesAllowed));
33              return;
34          }
35  
36          // classのアノテーションを確認
37          var clazz = resourceInfo.getResourceClass();
38          if (clazz.getAnnotation(DenyAll.class) != null) {
39              context.register(new DenyAllRequestFilter());
40              return;
41          }
42          rolesAllowed = clazz.getAnnotation(RolesAllowed.class);
43          if (rolesAllowed != null) {
44              context.register(new RolesAllowedRequestFilter(rolesAllowed));
45              return;
46          }
47          // 上に該当しないDenyAllもRolesAllowedも付いてないメソッドはどのロールでもアクセス可能
48      }
49  
50      private boolean enableFilter() {
51          var config = ConfigProvider.getConfig();
52          return config.getValue("jwt.filter.enable", Boolean.class);
53      }
54  }
