7   public interface MpConfigFactory {
8   
9       static final String MPCONFIG_FACTORY_CLASS_PROP = "mpconfig.factory.class";
10      static final String DEFAULT_FACTORY_CLASSNAME = "com.mamezou.rms.platform.config.helidon.HelidonMpConfigFactory";
11  
12      /**
13       * Since Config may be used before starting the CDI container, DI cannot be used,
14       * so an instance is dynamically created from the class name.
15       *
16       * @return MpConfigFactory instance
17       */
18      static MpConfigFactory newInstance() {
19          String factoryClassName = System.getProperty(MPCONFIG_FACTORY_CLASS_PROP, DEFAULT_FACTORY_CLASSNAME);
20          try {
21              return (MpConfigFactory) Class.forName(factoryClassName).getDeclaredConstructor().newInstance();
22          } catch (Exception e) {
23              throw new IllegalStateException(e);
24          }
25      }
26  
27      ConfigSource createFromExternalPath(String path);
28      ConfigSource createFromUrl(URL configUrl);
29  }
