3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.function.Predicate;
6   import java.util.stream.Collectors;
7   import java.util.stream.StreamSupport;
8   
9   import javax.enterprise.context.ApplicationScoped;
10  import javax.enterprise.context.Initialized;
11  import javax.enterprise.event.Observes;
12  import javax.inject.Inject;
13  
14  import org.eclipse.microprofile.config.Config;
15  
16  import lombok.extern.slf4j.Slf4j;
17  
18  /**
19   * Dump out Config at startup.
20   * Filter the output by setting configdump.filters as below.
21   * <pre>
22   * configdump:
23   *   filter-enable: true
24   *   filters:
25   *     - filter: server
26   *     - filter: security
27   * </pre>
28   */
29  @ApplicationScoped
30  @Slf4j(topic = "ConfigDump")
31  public class MpConfigDump {
32  
33      private Config config;
34  
35      @Inject
36      public MpConfigDump(Config config) {
37          this.config = config;
38      }
39      void onInialized(@Observes @Initialized(ApplicationScoped.class) Object event) {
40  
41          if (!log.isDebugEnabled()) {
42              return;
43          }
44  
45          List<String> filters = new ArrayList<>();
46          if (config.getOptionalValue("configdump.filter-enable", Boolean.class).orElse(false)) {
47              filters = StreamSupport.stream(config.getPropertyNames().spliterator(), false)
48                      .filter(s -> s.startsWith("configdump.filters"))
49                      .map(s -> config.getValue(s, String.class))
50                      .collect(Collectors.toList());
51          }
52  
53          Predicate<String> containsForwardMatch = new ContainsForwardMatch(filters);
54          String configDump = StreamSupport.stream(config.getPropertyNames().spliterator(), false)
55              .filter(containsForwardMatch)
56              .map(name -> name + "=" + config.getValue(name, String.class))
57              .sorted()
58              .collect(Collectors.joining(System.lineSeparator()));
59  
60          log.debug(System.lineSeparator() + configDump);
61      }
62  
63      static class ContainsForwardMatch implements Predicate<String> {
64  
65          private List<String> filters;
66          ContainsForwardMatch(List<String> filters) {
67              this.filters = filters;
68          }
69  
70          @Override
71          public boolean test(String name) {
72              return filters.isEmpty()
73                          ? true
74                          : filters.stream().anyMatch(filter -> name.startsWith(filter));
75          }
76      }
77  }
