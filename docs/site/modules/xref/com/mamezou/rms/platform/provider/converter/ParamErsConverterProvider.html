3   import java.lang.annotation.Annotation;
4   import java.lang.reflect.Type;
5   import java.time.LocalDate;
6   import java.time.format.DateTimeFormatter;
7   
8   import javax.ws.rs.ext.ParamConverter;
9   import javax.ws.rs.ext.ParamConverterProvider;
10  
11  import org.eclipse.microprofile.config.ConfigProvider;
12  
13  // Provider Class
14  // register by @RegisterProvider or Application#getClasseses()
15  public class ParamErsConverterProvider implements ParamConverterProvider {
16  
17      private final String pattern;
18  
19      public ParamErsConverterProvider() {
20          // ConfigがなぜかInjectで取得できないためProvierクラス経由で取得
21          this.pattern = ConfigProvider.getConfig().getValue("json.format.date", String.class);
22      }
23  
24      @SuppressWarnings("unchecked")
25      @Override
26      public <T> ParamConverter<T> getConverter(Class<T> rawType, Type genericType, Annotation[] annotations) {
27          if (rawType == LocalDate.class) {
28              return (ParamConverter<T>) new LocalDateConverter();
29          }
30          return null;
31      }
32  
33  
34      // ----------------------------------------------------- inner classes
35  
36      public class LocalDateConverter implements ParamConverter<LocalDate> {
37          @Override
38          public LocalDate fromString(String value) {
39              return LocalDate.parse(value, DateTimeFormatter.ofPattern(pattern));
40          }
41          @Override
42          public String toString(LocalDate value) {
43              //return value == null ? null : value.format(DateTimeFormatter.ofPattern(pattern));
44              return value == null ? "" : value.format(DateTimeFormatter.ofPattern(pattern));
45          }
46      }
47  }
