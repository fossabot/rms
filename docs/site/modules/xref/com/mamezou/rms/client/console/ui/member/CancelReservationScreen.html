17  import lombok.RequiredArgsConstructor;
18  
19  @RequiredArgsConstructor
20  public class CancelReservationScreen implements RmsScreen {
21  
22      private final RentalReservationClientApi clientApi;
23  
24      @Override
25      public Transition play(UserAccountClientDto loginUser, boolean printHeader) {
26  
27          if (printHeader) {
28              printScreenHeader(loginUser, "レンタル品予約キャンセル画面");
29          }
30  
31          var ownReservations = clientApi.getOwnReservations();
32  
33          // キャンセル可能な予約がない場合はメニューへ戻る
34          if (ownReservations.isEmpty()) {
35              printWarningInformation(CANNOT_CANCEL_RESERVATION_INFORMATION);
36              waitPressEnter();
37              return Transition.MEMBER_MAIN;
38          }
39  
40          // キャンセル可能な予約一覧を表示
41          println(CANCEL_RESERVATION_INFORMATION);
42          ownReservations.forEach(dto ->
43              println(ClientConstants.RESERVATION_FORMAT.format(dto))
44          );
45          blankLine();
46  
47          // キャンセルする予約を選択
48          var selectedReservation = newIntInputReader()
49                  .withSelectableValues(
50                          ownReservations.stream()
51                              .map(ReservationClientDto::getId)
52                              .collect(Collectors.toList()),
53                          SCREEN_BREAK_VALUE)
54                  .read("予約番号");
55          if (TextIoUtils.isBreak(selectedReservation)) {
56              return Transition.MEMBER_MAIN;
57          }
58  
59          blankLine();
60  
61          // レンタル品の予約キャンセルの実行
62          try {
63              clientApi.cancelReservation(selectedReservation);
64              printResultInformation(selectedReservation);
65              return Transition.MEMBER_MAIN;
66  
67          } catch (BusinessFlowClientException e) {
68              printServerError(e);
69              return play(loginUser, false); // start over!!
70  
71          }
72      }
73  
74      private void printResultInformation(Integer selectedItem) {
75          println("***** 予約キャンセル確定結果 *****");
76          printf("[%s]の予約をキャンセルしました", selectedItem);
77          blankLine();
78          blankLine();
79          waitPressEnter();
80      }
81  }
