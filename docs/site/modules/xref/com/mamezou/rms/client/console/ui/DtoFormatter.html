9   public interface DtoFormatter<T> {
10  
11      String format(T dto);
12  
13      static class RentalItemFormatter implements DtoFormatter<RentalItemClientDto> {
14          @Override
15          public String format(RentalItemClientDto dto) {
16              return String.format("[%s]%s シリアル番号：%s",
17                      dto.getId(),
18                      dto.getItemName(),
19                      dto.getSerialNo());
20          }
21      }
22  
23      static class ReservationFormatter implements DtoFormatter<ReservationClientDto> {
24          @Override
25          public String format(ReservationClientDto dto) {
26              return String.format("[%s] %s - %s %s %s %s",
27                      dto.getId(),
28                      DATETIME_FORMAT.format(dto.getStartDateTime()),
29                      DATETIME_FORMAT.format(dto.getEndDateTime()),
30                      dto.getRentalItemDto().getItemName(),
31                      dto.getUserAccountDto().getUserName(),
32                      dto.getNote());
33          }
34      }
35  
36      static class UserAccountFormatter implements DtoFormatter<UserAccountClientDto> {
37          @Override
38          public String format(UserAccountClientDto dto) {
39              return String.format("[%s] %s/%s %s %s",
40                      dto.getId(),
41                      dto.getLoginId(),
42                      dto.getPassword(),
43                      dto.getUserName(),
44                      dto.getUserType());
45          }
46      }
47  }
