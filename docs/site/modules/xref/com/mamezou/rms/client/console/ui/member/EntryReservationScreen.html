15  import com.mamezou.rms.client.console.ui.TransitionMap.Transition;
16  import com.mamezou.rms.client.console.ui.textio.TextIoUtils;
17  
18  import lombok.RequiredArgsConstructor;
19  
20  @RequiredArgsConstructor
21  public class EntryReservationScreen implements RmsScreen {
22  
23      private final RentalReservationClientApi clientApi;
24  
25      @Override
26      public Transition play(UserAccountClientDto loginUser, boolean printHeader) {
27  
28          if (printHeader) {
29              printScreenHeader(loginUser, "レンタル品予約画面");
30          }
31  
32          // レンタル品一覧を表示
33          var items = clientApi.getAllRentalItems();
34          println(ENTRY_RESERVATION_INFORMATION);
35          items.forEach(dto ->
36              println(RENATL_ITEM_FORMAT.format(dto))
37          );
38          blankLine();
39  
40          // 予約するレンタル品の選択
41          var selectedItem = newIntInputReader()
42                  .withSelectableValues(
43                          items.stream()
44                              .map(RentalItemClientDto::getId)
45                              .collect(Collectors.toList()),
46                          SCREEN_BREAK_VALUE)
47                  .read("レンタル品番号");
48          if (TextIoUtils.isBreak(selectedItem)) {
49              return Transition.MEMBER_MAIN;
50          }
51  
52          // 利用開始日時の入力
53          var startDateTime = newLocalDateTimeReader()
54                  .withFutureNow()
55                  .read("利用開始日時（入力例－2020/04/01 09:00）:");
56  
57          // 利用終了日時の入力
58          var endDateTime = newLocalDateTimeReader()
59                  .withFutureThan(startDateTime)
60                  .read("利用終了日時（入力例－2020/04/01 18:00）:");
61  
62          // 備考の入力
63          var note = newStringInputReader()
64                  .withMaxLength(15)
65                  .withDefaultValue("")
66                  .read("備考（空白可）");
67  
68          blankLine();
69  
70          // レンタル品予約の実行
71          try {
72              var addReservation = ReservationClientDto.ofTransient(startDateTime, endDateTime, note, selectedItem, loginUser.getId());
73              var newReservation = clientApi.addReservation(addReservation);
74              printResultInformation(newReservation);
75              return Transition.MEMBER_MAIN;
76  
77          } catch (BusinessFlowClientException e) {
78              printServerError(e);
79              return play(loginUser, false); // start over!!
80  
81          }
82      }
83  
84      private void printResultInformation(ReservationClientDto newReservation) {
85          println("***** 予約確定結果 *****");
86          printf(ClientConstants.RESERVATION_FORMAT.format(newReservation));
87          blankLine();
88          blankLine();
89          waitPressEnter();
90      }
91  }
