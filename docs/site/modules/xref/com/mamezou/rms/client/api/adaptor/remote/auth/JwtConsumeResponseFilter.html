3   import java.io.IOException;
4   
5   import javax.enterprise.inject.spi.CDI;
6   import javax.ws.rs.client.ClientRequestContext;
7   import javax.ws.rs.client.ClientResponseContext;
8   import javax.ws.rs.client.ClientResponseFilter;
9   import javax.ws.rs.core.HttpHeaders;
10  
11  import org.apache.commons.lang3.tuple.Pair;
12  
13  import com.mamezou.rms.client.api.login.JsonWebTokenConsumeEvent;
14  
15  /**
16   * レスポンスヘッダからJsonWebTokenを取得し通知を行うクラス
17   */
18  public class JwtConsumeResponseFilter implements ClientResponseFilter {
19  
20      @Override
21      public void filter(ClientRequestContext requestContext, ClientResponseContext responseContext) throws IOException {
22  
23          if (responseContext.getHeaders().containsKey(HttpHeaders.AUTHORIZATION)) {
24  
25              var notificator = CDI.current().select(JwtConsumeEventNotifyDelegator.class).get();
26              var value = responseContext.getHeaderString(HttpHeaders.AUTHORIZATION);
27              notificator.push(new JsonWebTokenConsumeEvent(Pair.of(HttpHeaders.AUTHORIZATION, value)));
28          }
29      }
30  }
