6   import java.lang.annotation.Documented;

10  import javax.validation.Constraint;
11  import javax.validation.Payload;
12  import javax.validation.constraints.NotNull;
13  import javax.validation.constraints.Size;
14  
15  /**
16   * パスワードチェックアノテーション。
17   * <pre>
18   * ・未入力でないこと
19   * ・5桁以上10桁以下であること
20   * </pre>
21   */
22  @Documented
23  @Constraint(validatedBy = {})
24  @Target({ METHOD, FIELD, ANNOTATION_TYPE, CONSTRUCTOR, PARAMETER })
25  @Retention(RUNTIME)
26  @NotNull // 空文字列はSizeで2重にエラーに掛かるのでNotBlankではなくNotNullにしている
27  @Size(min = 5, max = 10)
28  public @interface Passowrd {
29      String message() default "{com.mamezou.rms.service.contrains.Generic.message}";
30      Class<?>[] groups() default {};
31      Class<? extends Payload>[] payload() default {};
32  
33      @Target({ METHOD, FIELD, ANNOTATION_TYPE, CONSTRUCTOR, PARAMETER })
34      @Retention(RUNTIME)
35      @Documented
36      public @interface List {
37          Passowrd[] value();
38      }
39  }
