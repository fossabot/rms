1   package com.mamezou.rms.client.api.adaptor.remote.auth;
2   
3   import javax.enterprise.context.ApplicationScoped;
4   import javax.enterprise.event.Observes;
5   import javax.ws.rs.core.MultivaluedHashMap;
6   import javax.ws.rs.core.MultivaluedMap;
7   
8   import org.apache.commons.lang3.tuple.Pair;
9   import org.eclipse.microprofile.rest.client.ext.ClientHeadersFactory;
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  import com.mamezou.rms.client.api.login.JsonWebTokenConsumeEvent;
14  
15  /**
16   * サーバから発行されたJsonWebTokenをリクエストヘッダに付加するクラス
17   */
18  @ApplicationScoped
19  public class JwtClientHeadersFactory implements ClientHeadersFactory {
20  
21      private static final Logger LOG = LoggerFactory.getLogger(JwtClientHeadersFactory.class);
22      private Pair<String, String> authHeader;
23  
24      @Override
25      public MultivaluedMap<String, String> update(MultivaluedMap<String, String> incomingHeaders,
26              MultivaluedMap<String, String> clientOutgoingHeaders) {
27  
28          if (authHeader == null) {
29              return clientOutgoingHeaders;
30          }
31  
32          var newHeadersMap = new MultivaluedHashMap<String, String>(clientOutgoingHeaders);
33          newHeadersMap.add(authHeader.getKey(), authHeader.getValue());
34          return newHeadersMap;
35      }
36  
37      void onEvent(@Observes JsonWebTokenConsumeEvent event) {
38          LOG.info("ヘッダに追加するJWTを受信しました");
39          this.authHeader = event.getToken();
40      }
41  }
