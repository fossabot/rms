28  /**
29   * Acquire and release EntityManager in pre-processing and post-processing for each test.
30   * If a transaction is required, @TransactionalTest can be annotated so that the test method
31   * can multiply the transaction that is the transaction boundary.
32   * <pre>
33   * @ExtendWith(JpaTransactionalExtension.class)
34   * class RentalItemJpaRepositoryTest {
35   *
36   *   @BeforeEach
37   *   void setup(EntityManager em) {
38   *      ....
39   *   }
40   *   @TransactionalTest(shouldCommit = false)
41   *   void addTest() {
42   *      ....
43   *   }
44   * </pre>
45   */
46  public class JpaTransactionalExtension implements Extension,
47          BeforeAllCallback, AfterAllCallback, BeforeEachCallback, AfterEachCallback,
48          BeforeTestExecutionCallback, AfterTestExecutionCallback, ParameterResolver {
49  
50      private static final String CURRENT_ENTITY_FACTORY = "CURRENT_ENTITY_MANAGER_FACTORY";
51      private static final String CURRENT_ENTITY_MANAGER = "CURRENT_ENTITY_MANAGER";
52      private static final String CURRENT_ENTITY_TRANSACTION = "CURRENT_ENTITY_TRANSACTION";
53  
54  
55      // ----------------------------------------------------- before methods
56  
57      @Override
58      public void beforeAll(ExtensionContext context) {
59          var unitName = geTragetUnitName();
60          var properties = getPersistenceProperties(unitName);
61          var emf = Persistence.createEntityManagerFactory(unitName, properties);
62          getEntityManagerFactoryStore(context).put(CURRENT_ENTITY_FACTORY, new CloseableWrapper(emf));
63      }
64  
65      @Override
66      public void beforeEach(ExtensionContext context) throws Exception {
67          EntityManagerFactory emf = getEntityManagerFactoryStore(context).get(CURRENT_ENTITY_FACTORY, CloseableWrapper.class).unwrap();
68          getEntityManagerStore(context).put(CURRENT_ENTITY_MANAGER, new CloseableWrapper(emf.createEntityManager()));
69      }
70  
71      @Override
72      public boolean supportsParameter(ParameterContext parameterContext, ExtensionContext extensionContext) {
73          return parameterContext.getParameter().getType() == EntityManager.class;
74      }
75  
76      @Override
77      public Object resolveParameter(ParameterContext parameterContext, ExtensionContext extensionContext) {
78          return getEntityManagerStore(extensionContext).get(CURRENT_ENTITY_MANAGER, CloseableWrapper.class).unwrap();
79      }
80  
81      @Override
82      public void beforeTestExecution(ExtensionContext context) throws Exception {
83          if (!AnnotationSupport.isAnnotated(context.getTestClass(), TransactionalForTest.class)
84                  && !AnnotationSupport.isAnnotated(context.getTestMethod(), TransactionalForTest.class)) {
85              return;
86          }
87  
88          EntityManager em = getEntityManagerStore(context).get(CURRENT_ENTITY_MANAGER, CloseableWrapper.class).unwrap();
89          if (em == null) {
90              throw new IllegalStateException("EntityManage is unset.");
91          }
92  
93          var tx = em.getTransaction();
94          tx.begin();
95          getEntityManagerStore(context).put(CURRENT_ENTITY_TRANSACTION, tx);
96      }
97  
98  
99      // ----------------------------------------------------- after methods
100 
101     @Override
102     public void afterTestExecution(ExtensionContext context) {
103         // Give priority to Method Annotation
104         TransactionalForTest transactionalTest = AnnotationSupport
105                     .findAnnotation(context.getRequiredTestMethod(), TransactionalForTest.class)
106                 .orElse(AnnotationSupport.findAnnotation(context.getRequiredTestClass(), TransactionalForTest.class).orElse(null));
107         if (transactionalTest == null) {
108             return;
109         }
110 
111         var tx = getEntityManagerStore(context).remove(CURRENT_ENTITY_TRANSACTION, EntityTransaction.class);
112         if (transactionalTest.shouldCommit()) {
113             tx.commit();
114         } else {
115             tx.rollback();
116         }
117     }
118 
119     @Override
120     public void afterEach(ExtensionContext context) {
121         getEntityManagerStore(context).remove(CURRENT_ENTITY_MANAGER, CloseableWrapper.class).close();
122     }
123 
124     @Override
125     public void afterAll(ExtensionContext context) {
126         getEntityManagerFactoryStore(context).remove(CURRENT_ENTITY_FACTORY, CloseableWrapper.class).close();
127     }

130     // ----------------------------------------------------- private methods
131 
132     private String geTragetUnitName() {
133         return ConfigProvider.getConfig().getValue("test.db.connection.unitname", String.class);
134     }
135 
136     private Map<String, String> getPersistenceProperties(String unitName) {
137         var config = ConfigProvider.getConfig();
138         var keys = StreamSupport.stream(config.getPropertyNames().spliterator(), false)
139                 .filter(key -> key.startsWith("test.db.connection.properties."))
140                 .collect(Collectors.toList());
141         return keys.stream().collect(Collectors.toMap(
142                     key -> StringUtils.remove(key, "test.db.connection.properties."), // prop-key
143                     key -> config.getValue(key, String.class) // prop-value
144                 ));
145     }
146 
147     private Store getEntityManagerFactoryStore(ExtensionContext context) {
148         return context.getStore(Namespace.create(context.getRequiredTestClass()));
149     }
150 
151     private Store getEntityManagerStore(ExtensionContext context) {
152         return context.getStore(Namespace.create(getClass(), context.getRequiredTestMethod()));
153     }
154

156     // ----------------------------------------------------- private methods
157 
158     static class CloseableWrapper implements Store.CloseableResource {
159         private Object org;
160 
161         public CloseableWrapper(EntityManagerFactory closeable) {
162             this.org = closeable;
163         }
164         public CloseableWrapper(EntityManager closeable) {
165             this.org = closeable;
166         }
167 
168         @Override
169         public void close() {
170             if (org instanceof EntityManagerFactory) {
171                 var closeable = (EntityManagerFactory) org;
172                 if (closeable.isOpen()) {
173                     closeable.close();
174                 }
175             }
176             if (org instanceof EntityManager) {
177                 var closeable = (EntityManager) org;
178                 if (closeable.isOpen()) {
179                     closeable.close();
180                 }
181             }
182         }
183         @SuppressWarnings("unchecked")
184         public <T> T unwrap() {
185             return (T) org;
186         }
187     }
