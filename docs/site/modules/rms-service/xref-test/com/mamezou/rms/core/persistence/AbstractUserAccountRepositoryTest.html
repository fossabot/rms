18      @Test
19      void testGet() {
20          var expect = UserAccount.of(3, "admin", "admin", "管理者", "050-1111-2222", "連絡先3", UserType.ADMIN);

31      @Test
32      void testFindAll() {
33          var expected = List.of(
34                  UserAccount.of(1, "member1", "member1", "メンバー1", "070-1111-2222", "連絡先1", UserType.MEMBER),
35                  UserAccount.of(2, "member2", "member2", "メンバー2", "080-1111-2222", "連絡先2", UserType.MEMBER),
36                  UserAccount.of(3, "admin", "admin", "管理者", "050-1111-2222", "連絡先3", UserType.ADMIN)
37                  );
38          var actuals = repository().findAll();
39          assertThatToString(actuals).containsExactlyElementsOf(expected);
40      }

68      @Test
69      @TransactionalForTest
70      void testUpdate() {
71          var expect = UserAccount.of(1, "update1", "update2", "update3", "999-1111-2222", "update4", UserType.ADMIN);
72          var updateUser = repository().get(1);
73          updateUser.setLoginId("update1");
74          updateUser.setPassword("update2");
75          updateUser.setUserName("update3");
76          updateUser.setPhoneNumber("999-1111-2222");
77          updateUser.setContact("update4");
78          updateUser.setUserType(UserType.ADMIN);
79          var actual = repository().update(updateUser);
80          assertThatToString(actual).isEqualTo(expect);
81      }
