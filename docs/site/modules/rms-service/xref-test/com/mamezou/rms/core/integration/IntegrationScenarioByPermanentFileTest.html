20  @AddConfig(key = "csv.permanent.directory", value = IntegrationScenarioByPermanentFileTest.TEST_PERMANENT_DIR)

23  
24      public static final String TEST_PERMANENT_DIR = "./target/temp-integrationtest";
25  
26      @AfterAll
27      static void teardownAfterAll() throws Exception {
28          var targetDir = Paths.get(TEST_PERMANENT_DIR);
29          Files.list(targetDir).forEach(IntegrationScenarioByPermanentFileTest::deleteQuietly); // ファイル削除
30          Files.deleteIfExists(targetDir); // ディレクトリ削除
31      }
32  
33      static boolean deleteQuietly(Path path) {
34          try {
35              return Files.deleteIfExists(path);
36          } catch (IOException e) {
37              throw new IoSystemException(e);
38          }
39      }
40  
41      @Override
42      protected int expectedReregistrationId() {
43          return 4; // Fileはその時点のレコードのmax(reservation.id)+1となる
44      }
