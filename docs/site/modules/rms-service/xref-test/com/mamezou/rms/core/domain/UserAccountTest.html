196     @Test
197     void testPropetyValidation() {
198 
199         // エラーがないこと
200         UserAccount u = createAllOKUserAccount();
201         Set<ConstraintViolation<UserAccount>> result = validator.validate(u);
202         ConstraintViolationSetAssert.assertThat(result)
203             .hasNoViolations();
204 
205         // IDエラー
206         // -- グループ指定なし→未実行なのでエラーなし
207         u.setId(0);
208         result = validator.validate(u);
209         ConstraintViolationSetAssert.assertThat(result)
210             .hasNoViolations();
211         // -- グループ指定あり→バリデート実行でエラー
212         result = validator.validate(u, Update.class);
213         ConstraintViolationSetAssert.assertThat(result)
214             .hasSize(1)
215             .hasViolationOnPath("id")
216             .hasMessageEndingWith("Min.message");
217 
218         // ログインIDエラー(null)
219         u = createAllOKUserAccount();
220         u.setLoginId(null);
221         result = validator.validate(u);
222         ConstraintViolationSetAssert.assertThat(result)
223             .hasSize(1)
224             .hasViolationOnPath("loginId")
225             .hasMessageEndingWith("NotNull.message");
226 
227         // ログインIDエラー(空文字列)
228         u = createAllOKUserAccount();
229         u.setLoginId("");
230         result = validator.validate(u);
231         ConstraintViolationSetAssert.assertThat(result)
232             .hasSize(1)
233             .hasViolationOnPath("loginId")
234             .hasMessageEndingWith("Size.message");
235 
236 
237         // ログインIDエラー(5文字未満)
238         u = createAllOKUserAccount();
239         u.setLoginId("1234");
240         result = validator.validate(u);
241         ConstraintViolationSetAssert.assertThat(result)
242             .hasSize(1)
243             .hasViolationOnPath("loginId")
244             .hasMessageEndingWith("Size.message");
245 
246         // パスワードエラー(null)
247         u = createAllOKUserAccount();
248         u.setPassword(null);
249         result = validator.validate(u);
250         ConstraintViolationSetAssert.assertThat(result)
251             .hasSize(1)
252             .hasViolationOnPath("password")
253             .hasMessageEndingWith("NotNull.message");
254 
255         // パスワードエラー(空文字列)
256         u = createAllOKUserAccount();
257         u.setPassword("");
258         result = validator.validate(u);
259         ConstraintViolationSetAssert.assertThat(result)
260             .hasSize(1)
261             .hasViolationOnPath("password")
262             .hasMessageEndingWith("Size.message");
263 
264         // パスワードエラー(5文字未満)
265         u = createAllOKUserAccount();
266         u.setPassword("1234");
267         result = validator.validate(u);
268         ConstraintViolationSetAssert.assertThat(result)
269             .hasSize(1)
270             .hasViolationOnPath("password")
271             .hasMessageEndingWith("Size.message");
272 
273         // ユーザ名(null)
274         u = createAllOKUserAccount();
275         u.setUserName(null);
276         result = validator.validate(u);
277         ConstraintViolationSetAssert.assertThat(result)
278             .hasSize(1)
279             .hasViolationOnPath("userName")
280             .hasMessageEndingWith("NotBlank.message");
281 
282         // 電話番号(使用可能文字以外)
283         u = createAllOKUserAccount();
284         u.setPhoneNumber("12345%");
285         result = validator.validate(u);
286         ConstraintViolationSetAssert.assertThat(result)
287             .hasSize(1)
288             .hasViolationOnPath("phoneNumber")
289             .hasMessageEndingWith("PhoneNumberCharacter.message");
290 
291         // 電話番号(14文字より大きい)
292         u = createAllOKUserAccount();
293         u.setPhoneNumber("123456789012345");
294         result = validator.validate(u);
295         ConstraintViolationSetAssert.assertThat(result)
296             .hasSize(1)
297             .hasViolationOnPath("phoneNumber")
298             .hasMessageEndingWith("Size.message");
299 
300         // 連絡先(15文字より大きい)
301         u = createAllOKUserAccount();
302         u.setContact("1234567890123456");
303         result = validator.validate(u);
304         ConstraintViolationSetAssert.assertThat(result)
305             .hasSize(1)
306             .hasViolationOnPath("contact")
307             .hasMessageEndingWith("Size.message");
308 
309         // ユーザ区分(null)
310         u = createAllOKUserAccount();
311         u.setUserType(null);
312         result = validator.validate(u);
313         ConstraintViolationSetAssert.assertThat(result)
314             .hasSize(1)
315             .hasViolationOnPath("userType")
316             .hasMessageEndingWith("NotNull.message");
317     }
