9   /**
10   * 永続先に依らないリポジトリの共通操作
11   *
12   * @param <T> エンティティの型
13   */
14  public interface GenericRepository<T> {
15  
16      /**
17       * IDのエンティティを取得する。
18       *
19       * @param id ID
20       * @return エンティティ。該当なしはnull
21       */
22      T get(int id);
23  
24      /**
25       * 永続化されているエンティティを全件取得する
26       *
27       * @return エンティティの全件リスト。該当なしは空リスト
28       */
29      List<T> findAll();
30  
31      /**
32       * エンティティを追加する。
33       * 実装クラスもしくはメソッドに{@link ValidateParam}がアノテートすることでメソッド実行前に
34       * {@link Valid}によりオブジェクトのValidationが実行される。
35       *
36       * @param entity エンティティ
37       */
38      void add(T entity);
39  
40      /**
41       * コンフィグ定数
42       */
43      static class ApiType {
44          public static final String PROP_NAME ="persistence.apiType";
45          public static final String FILE = "file";
46          public static final String JPA = "jpa";
47      }
48  }
