12  
13  @ApplicationScoped
14  @EnabledIfRuntimeConfig(propertyName = ApiType.PROP_NAME, value = ApiType.FILE)
15  public class RentalItemFileRepository extends AbstractFileRepository<RentalItem> implements RentalItemRepository {
16  
17      @Inject
18      public RentalItemFileRepository(FileAccessor fileAccessor, EntityArrayConverter<RentalItem> converter) {
19          super(fileAccessor, converter);
20      }
21  
22      @Override
23      public RentalItem findBySerialNo(String serialNo) {
24          return this.load().stream()
25                  .filter(attributes -> attributes[1].equals(serialNo))
26                  .map(this.getConverter()::toEntity)
27                  .findFirst()
28                  .orElse(null);
29      }
30  }
