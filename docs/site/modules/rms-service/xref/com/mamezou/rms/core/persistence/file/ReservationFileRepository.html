5   import java.util.List;
6   import java.util.stream.Collectors;
7   
8   import javax.enterprise.context.ApplicationScoped;
9   import javax.inject.Inject;
10  
11  import com.mamezou.rms.core.domain.Reservation;
12  import com.mamezou.rms.core.domain.Reservation.DateTimePeriod;
13  import com.mamezou.rms.core.persistence.ReservationRepository;
14  import com.mamezou.rms.core.persistence.GenericRepository.ApiType;
15  import com.mamezou.rms.core.persistence.file.converter.EntityArrayConverter;
16  import com.mamezou.rms.core.persistence.file.io.FileAccessor;
17  import com.mamezou.rms.platform.extension.EnabledIfRuntimeConfig;
18  
19  @ApplicationScoped
20  @EnabledIfRuntimeConfig(propertyName = ApiType.PROP_NAME, value = ApiType.FILE)
21  public class ReservationFileRepository extends AbstractFileRepository<Reservation> implements ReservationRepository {
22  
23      @Inject
24      public ReservationFileRepository(FileAccessor fileAccessor, EntityArrayConverter<Reservation> converter) {
25          super(fileAccessor, converter);
26      }
27  
28      @Override
29      public List<Reservation> findByRentalItemAndStartDate(int rentalItemId, LocalDate startDate) {
30          return this.load().stream()
31                  .map(this.getConverter()::toEntity)
32                  .filter(reservation -> reservation.getStartDateTime().toLocalDate().equals(startDate))
33                  .filter(reservation -> reservation.getRentalItemId() == rentalItemId)
34                  .collect(Collectors.toList());
35      }
36  
37      @Override
38      public List<Reservation> findByReserverId(int reserverId) {
39          return this.load().stream()
40                  .map(this.getConverter()::toEntity)
41                  .filter(reservation -> reservation.getUserAccountId() == reserverId)
42                  .collect(Collectors.toList());
43      }
44  
45      @Override
46      public Reservation findOverlappedReservation(int rentalItemId, LocalDateTime startDateTime, LocalDateTime endDateTime) {
47          DateTimePeriod conditionOfPeriod = new Reservation.DateTimePeriod(startDateTime, endDateTime);
48          return this.load().stream()
49                  .map(this.getConverter()::toEntity)
50                  .filter(reservation -> reservation.getRentalItemId() == rentalItemId)
51                  .filter(reservation -> reservation.getReservePeriod().isOverlappedBy(conditionOfPeriod))
52                  .findFirst()
53                  .orElse(null);
54      }
55  
56      @Override
57      public void delete(Reservation reservation) {
58          this.delete(reservation.getId());
59      }
60  }
