26  @Dependent
27  @Slf4j
28  public class FileRepositoryInitializeProducers {
29  
30      // config value is "permanent" or "temporary"
31      private static final String FILE_TYPE_CONFIG_KEY = "csv.type";
32      // samaple key is "csv-file.permanent.file_name.reservation = reservation.csv"
33      private static final String FILE_NAME_CONFIG_KEY_FORMAT = "csv.%s.fileName.%s";
34  
35      private static final Map<Class<?>, String> FILE_NAME_TYPE_CONFIG_KEY_MAP = Map.of(
36                  RentalItemFileRepository.class, "rentalitem",
37                  ReservationFileRepository.class, "reservation",
38                  UserAccountFileRepository.class, "useraccount"
39              );
40  
41      @Inject
42      private Config config;
43  
44      @Produces
45      public FileAccessor creteFileAccessor(InjectionPoint injectionPoint) throws IOException {
46  
47          // 1%sの文字列取得
48          String fileType = config.getValue(FILE_TYPE_CONFIG_KEY, String.class);
49  
50          // 2%sの文字列取得
51          Class<?> beanClass = injectionPoint.getBean().getBeanClass();
52          String fileNameTypeConfigKey = FILE_NAME_TYPE_CONFIG_KEY_MAP.get(beanClass);
53  
54          // ファイル名のConfigKeyの決定
55          String fileNameConfigKey = String.format(FILE_NAME_CONFIG_KEY_FORMAT, fileType, fileNameTypeConfigKey);
56          String fileName = config.getValue(fileNameConfigKey, String.class);
57  
58          // フィルパスの取得
59          Path filePath;
60          switch (fileType) {
61              case "permanent":
62                  filePath = new PathResolver.FixedDirPathResolver().resolve(fileName);
63                  break;
64              case "temporary":
65                  filePath =  FileAccessor.copyResourceToRealPath(fileName, new PathResolver.TempDirPathResolver());
66                  break;
67              default:
68                  throw new IllegalArgumentException("unknown fileType -> " + fileType);
69          }
70          log.info("[{}]モードでファイルをオープンしました。PATH={}", fileType, filePath);
71  
72          return new FileAccessor(filePath);
73      }
74  
75      @Produces
76      public EntityArrayConverter<RentalItem> createRentalItemConverter(InjectionPoint injectionPoint) {
77          return RentalItemArrayConverter.INSTANCE;
78      }
79  
80      @Produces
81      public EntityArrayConverter<Reservation> createReservationConverter(InjectionPoint injectionPoint) {
82          return ReservationArrayConverter.INSTANCE;
83      }
84  
85      @Produces
86      public EntityArrayConverter<UserAccount> createUserAccoutConverter(InjectionPoint injectionPoint) {
87          return UserAccountArrayConverter.INSTANCE;
88      }
89  }
