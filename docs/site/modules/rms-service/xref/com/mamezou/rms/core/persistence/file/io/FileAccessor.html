23  /**
24   * ファイルアクセスクラス
25   */
26  public class FileAccessor {
27  
28      /** ファイルパス */
29      private Path filePath;
30  
31      // ----------------------------------------------------- constructor methods
32  
33      /**
34       * コンストラクタ
35       * <p>
36       * @param csvFilePath ファイルパス
37       */
38      public FileAccessor(Path csvFilePath) {
39          this.filePath = csvFilePath;
40      }
41  
42      // ----------------------------------------------------- public methods
43  
44      /**
45       * ファイルパスを取得する。
46       * <p>
47       * @return ファイルパス
48       */
49      public Path getFilePath() {
50          return filePath;
51      }
52  
53      /**
54       * ファイルを読み込む。
55       * <p>
56       * @param dataList 読み込んだデータを埋めて返す(in/out)
57       * @return 読み込み件数
58       * @throws IOException 読み込みエラーが発生した場合
59       */
60      public int load(List<String[]> dataList) throws IOException {
61          @Cleanup CSVParser parser = CSVParser.parse(filePath, StandardCharsets.UTF_8, CSVFormat.RFC4180);
62          parser.getRecords().stream()
63                  .map(record -> StreamSupport.stream(record.spliterator(), false).collect(Collectors.toList()))
64                  .map(values -> {
65                      var array = new String[values.size()];
66                      values.toArray(array);
67                      return array;
68                  })
69                  .forEach(dataList::add);
70          return dataList.size();
71      }
72  
73      /**
74       * ファイルに書き込む。
75       * <p>
76       * @param targetData 書き込みデータ
77       * @throws IOException 読み込みエラーが発生した場合
78       */
79      public void save(String[] targetData) throws IOException {
80          List<String> singleLine = new ArrayList<>();
81          singleLine.add(CSVFormat.RFC4180.format((Object[])targetData));
82          Files.write(filePath, singleLine, StandardCharsets.UTF_8, WRITE, APPEND);
83      }
84  
85      /**
86       * 全件をファイルに書き込む。
87       * ファイルに既にあるデータは削除される。
88       *
89       * @param allData 書き込みデータ
90       * @throws IOException 読み込みエラーが発生した場合
91       */
92      public void saveAll(List<String[]> allData) throws IOException {
93          Stream<CharSequence> allLines = allData.stream()
94                  .map(items -> CSVFormat.RFC4180.format((Object[]) items)); // Memory-friendly and lazy stringification
95          Files.write(filePath, allLines::iterator, StandardCharsets.UTF_8, CREATE, TRUNCATE_EXISTING);
96      }
97  
98      /**
99       * 指定されたリソースファイルを一時ディレクトリにコピーする。
100      * <p>
101      * @param resourcePath リソースファイル
102      * @param resolver コピー先の一時ディレクトリが指定されたPathResolver
103      * @return 一時ディレクトリにコピーされたファイルのパス
104      * @throws IOException ファイル入出力エラーが発生した場合
105      */
106     public static Path copyResourceToRealPath(String resourcePath, PathResolver resolver) {
107         String[] resourcePathNodes = resourcePath.split("/");
108         String outputFileName = resourcePathNodes[resourcePathNodes.length - 1];
109         return copyResourceToRealPath(resourcePath, resolver, outputFileName);
110     }
111 
112     public static Path copyResourceToRealPath(String resourcePath, PathResolver resolver, String outputFileName) {
113         try (InputStream in = FileAccessor.class.getResourceAsStream("/" + resourcePath)) {
114             if (!Files.exists(resolver.getBaseDir())) {
115                 Files.createDirectory(resolver.getBaseDir());
116             }
117             Path outputFilePath = resolver.resolve(outputFileName);
118             Files.copy(in, outputFilePath);
119             return outputFilePath;
120         } catch (IOException e) {
121             throw new IoSystemException(e);
122         }
123     }

125 }
