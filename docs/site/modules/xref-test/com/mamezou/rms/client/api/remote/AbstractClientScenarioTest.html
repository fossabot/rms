40      @Test
41      @Order(1)
42      void testAddUserAccount() {
43          // -----------------------------------
44          // ユーザの登録
45          // -----------------------------------
46          // ★:ADMINから開始
47          service().authenticate("admin", "admin");
48  
49          UserAccountClientDto newUser = UserAccountClientDto.
50                  ofTransient("testUser", "userTest", "テスト太郎", "1234", "なぞなぞ部門", ClientUserType.MEMBER);
51          UserAccountClientDto addedUser = service().addUserAccount(newUser);
52  
53          // 登録を依頼した内容と登録された内容が同じか？
54          // 登録したユーザに発番された番号が既存のレコード数(3) +1であること
55          // その他のユーザ情報が登録したユーザの情報と同じか
56          newUser.setId(3 + 1);
57          assertThatToString(addedUser).isEqualTo(newUser);
58  
59          // -----------------------------------
60          // 登録したユーザでログインできるか
61          // -----------------------------------
62          // ★:MEMBERに切り替え
63          UserAccountClientDto loginUser = service().authenticate(addedUser.getLoginId(), addedUser.getPassword());
64  
65          // 取得したログインユーザの情報が登録したユーザの情報と同じであること
66          assertThat(loginUser).isNotNull();
67          assertThatToString(loginUser).isEqualTo(addedUser);
68  
69          // -----------------------------------
70          // 登録したユーザを再登録（重複エラー）
71          // -----------------------------------
72          // ★:ADMINに切り替え
73          service().authenticate("admin", "admin");
74          // 名前が重複
75          catchThrowableOfType(
76                  () -> service().addUserAccount(addedUser),
77                  BusinessFlowClientException.class
78              );
79  
80          // @Order(3)の事前条件として設定
81          context.loginUser = loginUser;
82      }

112     @Test
113     @Order(3)
114     void testAddReservation() {
115 
116         // -----------------------------------
117         // @Order(1)で登録したログインユーザと@Order(2)で登録したレンタル品を予約できるか
118         // -----------------------------------
119 
120         // 事前条件の取得
121         UserAccountClientDto loginUser = this.context.loginUser;
122         RentalItemClientDto addedItem = this.context.addedRentalItem;
123 
124         // ★:MEMBERに切り替え
125         service().authenticate(loginUser.getLoginId(), loginUser.getPassword());
126 
127         // 登録したレンタル品がレンタル品一覧に含まれていること
128         List<RentalItemClientDto> items = service().getAllRentalItems();
129         RentalItemClientDto selectedItem = items.stream()
130                 .filter(item -> item.getId() == addedItem.getId())
131                 .findFirst()
132                 .orElse(null);
133         assertThat(selectedItem).isNotNull();
134 
135         // 登録したユーザとレンタル品で予約
136         var startDateTime = LocalDateTime.now().plusDays(1).truncatedTo(MINUTES);
137         var endDateTime = startDateTime.plusDays(1);
138         var note = "備考";
139         var newReservation = ReservationClientDto.ofTransient(startDateTime, endDateTime, note, addedItem.getId(), loginUser.getId());
140         var addedReservation = service().addReservation(newReservation);
141 
142         newReservation.setId(3 + 1);
143         newReservation.setRentalItemDto(addedItem);
144         newReservation.setUserAccountDto(loginUser);
145         assertThatToString(addedReservation).isEqualTo(newReservation);
146 
147         // -----------------------------------
148         // 予約照会で登録した予約を参照できるか
149         // -----------------------------------
150         var reservations = service().findReservationByRentalItemAndStartDate(
151                 addedReservation.getRentalItemId(),
152                 addedReservation.getStartDateTime().toLocalDate());
153 
154         // 該当の予約が1件であること
155         assertThat(reservations).hasSize(1);
156         // 該当の予約が登録した予約と同じであることaddedReservation.toString()
157         assertThatToString(reservations.get(0)).isEqualTo(addedReservation);
158     }

160     @Test
161     @Order(4)
162     void testCancelReservation() {
163         // -----------------------------------
164         // @Order(3)で登録した予約をキャンセルできるか
165         // -----------------------------------
166         // 事前条件の取得と設定
167         var loginUser = context.loginUser;
168 
169         // ログインユーザの予約一覧を取得
170         //var ownReservations = service().findReservationByReserverId(loginUser.getId());
171         var ownReservations = service().getOwnReservations();
172         // 該当の予約が1件であること
173         assertThat(ownReservations).hasSize(1);
174 
175         // ログインユーザ以外の予約を削除
176         catchThrowableOfType(
177             () -> service().cancelReservation(1),
178             RentalReservationClientException.class
179         );
180 
181         // キャンセル対象の予約を取得
182         var cancelTarget = ownReservations.get(0);
183         // キャンセルの実行
184         service().cancelReservation(cancelTarget.getId());
185 
186         // 再度ログインユーザの予約一覧を取得
187         ownReservations = service().findReservationByReserverId(loginUser.getId());
188         // 自分の予約一覧に出てなこないこと
189         assertThat(ownReservations).isEmpty();
190     }
